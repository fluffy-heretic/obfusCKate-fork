		if = {
			limit = {

			}
			scope:owner = {
				send_interface_toast = {
					title = sway_invalidated_title

				}
			}
		}
		else_if = {	
			limit = {
				scope:owner = {
					is_at_war_with = scope:target
				}
			}
			scope:owner = {
				send_interface_toast = {
					title = sway_invalidated_title
					left_icon = scope:target
					custom_tooltip = sway_invalidated_war
				}
			}
		}
 
		if = {
			limit = {
				scope:target = {  
					NOT = { in_diplomatic_range = scope:owner } 
				}
			}  
			scope:owner = {
				send_interface_toast = {

					left_icon = scope:target 
					custom_description_no_bullet = {
